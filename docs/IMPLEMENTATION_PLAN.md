```python
from prompt_opt_env.server.actions import (
    add_context, shorten, add_example, rephrase, add_constraint, apply_action

    result = add_context("Tell me about dogs.", "Dogs are domesticated mammals.")
    assert "Context: Dogs are domesticated mammals." in result

def test_add_context_no_duplicate():
```

```python
from unittest.mock import patch, MagicMock
from prompt_opt_env.server.grader import Grader

def test_rouge_score_returns_float():
    grader = Grader(grader_type="rouge")
    score, output = grader.score("Explain recursion.", "Recursion calls itself.", task_id=14)
    assert isinstance(score, float)
    assert 0.0 <= score <= 1.0

def test_rouge_score_better_match_higher():
    grader = Grader(grader_type="rouge")
    reference = "Recursion is when a function calls itself."
    # dummy output for task 14 is close to reference
    score14, _ = grader.score("prompt", reference, task_id=14)
    # dummy output for task 0 is unrelated
    score0, _ = grader.score("prompt", reference, task_id=0)
    assert score14 >= score0

def test_clip_reward():
    assert Grader.clip_reward(1.5) == 1.5  # within [−1, 2]
    assert Grader.clip_reward(-2.0) == -1.0  # clipped at −1
```

```python
from prompt_opt_env.server.prompt_opt_environment import PromptOptEnvironment

    env = PromptOptEnvironment()
```

```python
import asyncio
from prompt_opt_env import PromptOptEnv, PromptAction
import random

async def random_agent_demo():
    """Demo: random agent runs one episode."""
    async with PromptOptEnv(base_url="wss://YOUR_USERNAME-prompt-opt-env.hf.space") as env:
        result = await env.reset()
        print(f"Task: {result.observation.task_description}")
        total_reward = 0.0
        while not result.done:
            action = PromptAction(action_id=random.randint(0, 4))
            result = await env.step(action)
            total_reward += result.reward
            print(f"Step {result.observation.step_count}: {result.observation.info['action_applied']} → reward={result.reward:.3f}")
        print(f"Episode done. Total reward: {total_reward:.3f}")

asyncio.run(random_agent_demo())
```

## Round 2 Extensions (post-shortlisting, April 18–26)

Prepare these during the bootcamp (Apr 18–19):

### E1: LLM-as-judge grader
Replace ROUGE with a second LLM call that judges output quality on a 1–10 scale. Reward = judge_score / 10.

### E2: Multi-task episodes
Instead of one task per episode, chain 3 tasks. Agent must optimise prompts for all 3. Episode reward = mean across all tasks.

### E3: Chain-of-thought action
Add `action_id=5: ADD_COT` — appends "Let's think step by step:" to prompt. Teaches agent that CoT improves LLM outputs.

### E4: Curriculum difficulty
Start with easy tasks (task_ids 0,1,4,7,8). Unlock medium tasks when agent hits mean episode reward > 0.4. Unlock hard tasks at > 0.6.

### E5: Multi-turn optimisation
Allow agent to see the LLM's actual output, not just the score, and use it to decide next action.
